     emergency_id uuid references emergency_alerts(id),
     written_by uuid references profiles(id),
     written_by_name text,
     written_by_role text,
     note text not null,
     created_at timestamp default now()
   );

   create policy "Authenticated users can manage emergency notes"

   using (auth.role() = 'authenticated');
